def hc_post(self, x: torch.Tensor, residual: torch.Tensor, h_post: torch.Tensor, h_res: torch.Tensor):

        \# x: [b, s, D], residual: [b, s, n, D], h_post: [b, s, n], h_res: [b, s, n, n], y:[b, s, n, D]

        y = (h_post.unsqueeze(-1) * x.unsqueeze(-2) +
